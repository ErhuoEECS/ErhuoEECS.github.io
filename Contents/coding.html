  • three-sum

    Given an array nums of n integers, are there elements a, b, c in nums such that a + b + c = 0? Find all unique triplets in the array which gives the sum of zero. Notice that the solution set must not contain duplicate triplets. 

    Example 1                    
                    
    Input: nums = [-1,0,1,2,-1,-4]
    Output: [[-1,-1,2],[-1,0,1]]
                    
                
    Constraints:
                         
    0 <= nums.length <= 3000
    -105 <= nums[i] <= 105
                
                
                        
#include <vector>
#include <algorithm>
using namespace std;
class Solution
{
public:
    vector<vector<int>> AnsVector;
    vector<vector<int>> threeSum(vector<int> &nums)
    {
        sort(nums.begin(), nums.end());
        vector<int> temp;
        for (int left = 0; left < nums.size(); left++)
        {
            if (left == 0 || (left > 0 && nums[left] != nums[left - 1]))
            {
                int middle = left + 1, right = nums.size() - 1;
                while (middle < right)
                {
                    if (nums[middle] + nums[right] < -nums[left])
                        middle++;
                    else if (nums[middle] + nums[right] > -nums[left])
                        right--;
                    else if (nums[middle] + nums[right] == -nums[left])
                    {
                        int middleNums = nums[middle];
                        AnsVector.push_back({nums[left],middleNums,nums[right]});
                        temp.clear();
                        while (middleNums == nums[middle] && middle < right)
                        {
                            middle++;
                        }
                    }
                }
            }
        }
        return AnsVector;
    }
};
